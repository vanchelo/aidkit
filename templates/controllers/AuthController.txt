<?php

class AidkitAuthController extends AidkitController {

	// Show the Login Mask
	public function getIndex()
	{
		return View::make('aidkit::'.Config::get('aidkit::config.login'));
	}

	public function postIndex()
	{
		$remember = false;
		
		if(Input::has('remember')) $remember = true;

		if(Auth::attempt(array('username' => Input::get('username'),'password'=>Input::get('password')),$remember))
		{
			Event::fire('aidkit.login', array(Auth::user()));
			return Redirect::route('dashboard');
		} 

		return View::make('aidkit::'.Config::get('aidkit::config.login'))->with('status','failed');
	}

	public function logout()
	{
		
		Auth::logout(); 
		Session::flush();

		return Redirect::to('/');

	}
}
